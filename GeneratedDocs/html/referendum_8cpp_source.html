378 uint8_t referendum::calculateStatus(name referendum_id, name dac_id) {
379  referenda_table referenda(get_self(), dac_id.value);
380  auto ref = referenda.find(referendum_id.value);
381  check(ref != referenda.end(), "ERR:REFERENDUM_NOT_FOUND::Referendum not found");
382 
383  config_item config = config_item::get_current_configs(get_self(), dac_id);
384  uint64_t quorum = 0, current_yes = 0, current_no = 0, current_abstain = 0, current_all = 0;
385  uint16_t pass_rate = config.pass[ref->type]; // integer with 2 decimals
386  uint8_t status = referendum_status::STATUS_OPEN;
387  uint32_t time_now = current_time_point().sec_since_epoch();
388  uint64_t yes_percentage = 0;
389 
390  map<uint8_t, uint64_t> votes;
391 
392  if (ref->expires.sec_since_epoch() < time_now) {
393  if (ref->voting_type == count_type::COUNT_TOKEN) {
394  quorum = config.quorum_token[ref->type];
395  votes = ref->token_votes;
396  } else {
397  quorum = config.quorum_account[ref->type];
398  votes = ref->account_votes;
399  }
400 
401  uint64_t total = 0;
402  for (auto v : votes) {
403  total += v.second;
404  }
405 
406  current_yes = votes.at(vote_choice::VOTE_YES);
407  current_no = votes.at(vote_choice::VOTE_NO);
408  current_abstain = votes.at(vote_choice::VOTE_ABSTAIN);
409  current_all = current_yes + current_no + current_abstain;
410 
411  // check we have made quorum
412  double yes_percentage = 0.0;
413  if (total >= quorum) {
414  // quorum has been reached, check we have passed
415  yes_percentage =
416  (double(current_yes) / double(current_all)) * 10000.0; // multiply by 10000 to get integer with 2 dp
417  pass_rate = config.pass[ref->type];
418  if (uint64_t(yes_percentage) >= pass_rate) {
419  status = referendum_status::STATUS_PASSING;
420  } else {
421  status = referendum_status::STATUS_FAILING;
422  }
423  } else {
424  status = referendum_status::STATUS_QUORUM_NOT_MET;
425  }
426 
427  print("referendum id : ", referendum_id, ", dac id : ", dac_id, ", quorum : ", quorum,
428  ", pass rate : ", pass_rate, ", current rate : ", current_all, ", total : ", total, ", yes : ", current_yes,
429  "double: ", double(current_yes), "all double: ", double(current_all), ", yes% : ", yes_percentage,
430  " status: ", status);
431  }
432 
433  return status;
434 }
