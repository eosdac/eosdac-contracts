367  if (existing_staketime != staketimes.end()) {
368  unstake_delay = existing_staketime->delay;
369  }
370  uint32_t release_time = current_time_point().sec_since_epoch() + unstake_delay;
371 
372  uint64_t next_id = unstakes.available_primary_key();
373  unstakes.emplace(account, [&](unstake_info &u) {
374  u.key = next_id;
375  u.account = account;
376  u.stake = quantity;
377  u.release_time = time_point_sec(release_time);
378  });
379 
380  // notify of stake delta
381  send_stake_notification(account, -quantity, dac);
382 
383  // Remove from stake

385 
386  // deferred transaction to refund
387  transaction trx;

389  permission_level{get_self(), "notify"_n}, get_self(), "refund"_n, make_tuple(next_id, quantity.symbol)));
390  trx.delay_sec = unstake_delay;

394  void eosdactokens::staketime(name account, uint32_t unstake_time, symbol token_symbol) {
395  require_auth(account);
396 
397  dacdir::dac dac = dacdir::dac_for_symbol(extended_symbol{token_symbol, get_self()});
398  stake_config config = stake_config::get_current_configs(get_self(), dac.dac_id);
399  check(config.enabled, "ERR::STAKING_NOT_ENABLED::Staking is not enabled for this token");
400  staketimes_table staketimes(get_self(), dac.dac_id.value);
401  stakes_table stakes(get_self(), dac.dac_id.value);
402  unstakes_table unstakes(get_self(), dac.dac_id.value);
403  auto unstakes_idx = unstakes.get_index<"byaccount"_n>();
404 
405  check(unstake_time <= config.max_stake_time, "ERR::TIME_GREATER_MAX::Unstake time is greater than the maximum");
406  check(unstake_time >= config.min_stake_time, "ERR::TIME_LESS_MIN::Unstake time is less than the minimum");
407 
408  auto existing_stake = stakes.find(account.value);
409  auto existing_unstake = unstakes_idx.find(account.value);
410  auto existing_time = staketimes.find(account.value);
411  if ((existing_stake != stakes.end() || existing_unstake != unstakes_idx.end()) &&
412  existing_time != staketimes.end()) {
413  check(existing_time->delay <= unstake_time,
414  "ERR::CANNOT_REDUCE_STAKE_TIME::You cannot reduce the stake time if you have tokens staked or in the process of unstaking");
415  }
416 
417  uint32_t current_unstake_time = config.min_stake_time;
418  if (existing_time == staketimes.end()) {
419  staketimes.emplace(account, [&](staketime_info &s) {
420  s.account = account;
421  s.delay = unstake_time;
422  });
423  } else {
424  current_unstake_time = existing_time->delay;
425  staketimes.modify(*existing_time, account, [&](staketime_info &s) {
426  s.delay = unstake_time;
427  });
428  }
429 
430  // send notification for unstake at current delay and then stake with new delay
431  name custodian_contract = dac.account_for_type(dacdir::CUSTODIAN);
432  name vote_contract = dac.account_for_type(dacdir::VOTE_WEIGHT);
433  name notify_contract = (vote_contract) ? vote_contract : custodian_contract;
434  asset current_stake = eosdac::get_staked(account, get_self(), token_symbol);
435 
436  vector<account_stake_delta> stake_deltas_sub = {{account, -current_stake, current_unstake_time}};
437  action(permission_level{get_self(), "notify"_n}, notify_contract, "stakeobsv"_n,
438  make_tuple(stake_deltas_sub, dac.dac_id))
439  .send();
440 
441  vector<account_stake_delta> stake_deltas_add = {{account, current_stake, unstake_time}};
442  action(permission_level{get_self(), "notify"_n}, notify_contract, "stakeobsv"_n,
443  make_tuple(stake_deltas_add, dac.dac_id))
444  .send();
445  }

528  unstake_delay = existing_staketime->delay;

534  .send();
535 
536  if (referendum_contract && is_account(referendum_contract)) {
537  action(permission_level{get_self(), "notify"_n}, referendum_contract, "stakeobsv"_n,
