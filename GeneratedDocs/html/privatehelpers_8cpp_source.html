5 void daccustodian::updateVoteWeight(name custodian, int64_t weight, name dac_id) {
6  if (weight == 0) {
7  print("\n Vote has no weight - No need to continue.");
8  return;
9  }
10  candidates_table registered_candidates(_self, dac_id.value);
11 
12  auto candItr = registered_candidates.find(custodian.value);
13  if (candItr == registered_candidates.end()) {
14  eosio::print("Candidate not found while updating from a transfer: ", custodian);
15  return; // trying to avoid throwing errors from here since it's unrelated to a transfer action.?!?!?!?!
16  }
17 
18  if (weight < 0) {
19  check((candItr->total_votes + weight) < candItr->total_votes, "Underflow in updateVoteWeight");
20  } else {
21  check((candItr->total_votes + weight) >= candItr->total_votes, "Overflow in updateVoteWeight");
22  }
23 
24  registered_candidates.modify(candItr, same_payer, [&](auto &c) {
25  c.total_votes += weight;
26  eosio::print("\nchanging vote weight: ", custodian, " by ", weight);
27  });
28 }

70 void daccustodian::modifyVoteWeights(int64_t vote_weight, vector<name> oldVotes, vector<name> newVotes, name dac_id) {
71  // This could be optimised with set diffing to avoid remove then add for unchanged votes. - later
72 
73  if (vote_weight == 0) {

78 
79  // New voter -> Add the tokens to the total weight.
80  if (vote_weight > 0) {

85  "Underflow in total_weight_of_votes");
86  }
87 
88  if (oldVotes.size() == 0)
89  currentState.total_weight_of_votes += vote_weight;
90 
91  // Leaving voter -> Remove the tokens to the total weight.
92  if (newVotes.size() == 0)
93  currentState.total_weight_of_votes -= vote_weight;
94 
95  currentState.save(get_self(), dac_id);
96 
97  updateVoteWeights(oldVotes, -vote_weight, dac_id);
98  updateVoteWeights(newVotes, vote_weight, dac_id);
